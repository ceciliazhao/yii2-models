 19 /**
 20  * 一个模型的某个属性可能对应多个责任者，该 trait 用于处理此种情况。此种情况违反
 21  * 了关系型数据库第一范式，因此此 trait 只适用于责任者属性修改不频繁的场景，在开
 22  * 发时必须严格测试数据一致性，并同时考量性能。
 23  * Basic Principles:
 24  * <ol>
 25  * <li>when adding blame, it will check whether each of blames including to be
 26  * added is valid.
 27  * </li>
 28  * <li>when removing blame, as well as counting, getting or setting list of them,
 29  * it will also check whether each of blames is valid.
 30  * </li>
 31  * <li>By default, once blame was deleted, the guid of it is not removed from
 32  * list of blames immediately. It will check blame if valid when adding, removing,
 33  * counting, getting and setting it. You can define a blame model and attach it
 34  * events triggered when inserting, updating and deleting a blame, then disable
 35  * checking the validity of blames.
 36  * </li>
 37  * </ol>
 38  * Notice:
 39  * <ol>
 40  * <li>You must specify two properties: $multiBlamesClass and $multiBlamesAttribute.
 41  * <ul>
 42  * <li>$multiBlamesClass specify the class name of blame.</li>
 43  * <li>$multiBlamesAttribute specify the field name of blames.</li>
 44  * </ul>
 45  * </li>
 46  * <li>You should rename the following methods to be needed optionally.</li>
 47  * </ol>
 48  * @property-read array $multiBlamesAttributeRules
 49  * @property string[] $blameGuids
 50  * @property-read array $allBlames
 51  * @property-read array $nonBlameds
 52  * @property-read integer $blamesCount

 98     /**
 99      * Add specified blame.
100      * @param [multiBlamesClass]|string $blame
101      * @return false|array
102      */
103     public function addBlame($blame)
104     {
105         if (!is_string($this->multiBlamesAttribute)) {
106             return false;
107         }
108         $blameGuid = '';
109         if (is_string($blame)) {
110             $blameGuid = $blame;
111         }
112         if ($blame instanceof $this->multiBlamesClass) {
113             $blameGuid = $blame->guid;
114         }
115         $blameGuids = $this->getBlameGuids(true);
116         if (array_search($blameGuid, $blameGuids)) {
117             throw new \yii\base\InvalidParamException('the blame has existed.');
118         }
119         if ($this->getBlamesCount() >= $this->blamesLimit) {
120             throw new \yii\base\InvalidCallException("the limit($this->blamesLimit) of blames has been reached.");
121         }
122         $blameGuids[] = $blameGuid;
123         $this->setBlameGuids($blameGuids);
124         return $this->getBlameGuids();
125     }

144     /**
145      * Add specified blame, or create it before adding if doesn't exist.
146      * But you should save the blame instance before adding, or the operation
147      * will fail.
148      * @param [[$multiBlamesClass]]|string|array $blame 
149      * It will be regarded as blame's guid if it is a string. And assign the
150      * reference parameter $blame the instance if it existed, or create one if not
151      * found.
152      * If it is [[$multiBlamesClass]] instance and existed, then will add it, or
153      * false will be given if it is not found in database. So if you want to add
154      * blame instance, you should save it before adding.
155      * If it is a array, it will be regarded as configuration array of blame.
156      * Notice! This parameter passed by reference, so it must be a variable!
157      * @param \vistart\Models\models\BaseUserModel $user whose blame.
158      * If null, it will take this blameable model's user.
159      * @return false|array false if blame created failed or not enable this feature.
160      * blames guid array if created and added successfully.
161      * @throws \yii\base\InvalidConfigException
162      * @throws \yii\base\InvalidParamException
163      * @see addBlame()
164      */
165     public function addOrCreateBlame(&$blame = null, $user = null)
166     {
167         if (!is_string($this->multiBlamesClass)) {
168             throw new \yii\base\InvalidConfigException('$multiBlamesClass must be specified if you want to use multiple blameable features.');
169         }
170         if (is_array($blame)) {
171             if ($user == null) {
172                 $user = $this->user;
173             }
174             $blame = static::getOrCreateBlame($blame, $user);
175             if (!$blame->save()) {
176                 return false;
177             }
178             return $this->addBlame($blame->guid);
179         }
180         $blameGuid = '';
181         if (is_string($blame)) {
182             $blameGuid = $blame;
183         }
184         if ($blame instanceof $this->multiBlamesClass) {
185             $blameGuid = $blame->guid;
186         }
187         if (($mbi = static::getBlame($blameGuid)) !== null) {
188             return $this->addBlame($mbi);
189         }
190         return false;
191     }

264     /**
265      * Remove invalid blame guid from provided guid array.
266      * @param array $guids guid array of blames.
267      * @return array guid array of blames unset invalid.
268      */
269     protected function unsetInvalidBlames($guids)
270     {
271         $checkedGuids = Number::unsetInvalidUuids($guids);
272         $multiBlamesClass = $this->multiBlamesClass;
273         foreach ($checkedGuids as $key => $guid) {
274             $blame = $multiBlamesClass::findOne($guid);
275             if (!$blame) {
276                 unset($checkedGuids[$key]);
277             }
278         }
279         $diff = array_diff($guids, $checkedGuids);
280         $eventName = static::$eventMultipleBlamesChanged;
281         $this->trigger($eventName, new MultipleBlameableEvent(['blamesChanged' => !empty($diff)]));
282         return $checkedGuids;
283     }
