 33      * <Sub Domain Name> => [
 34      *     'component' => <URL Manager Component Configuration Array>,
 35      *     'scheme' => 'http'(default) or 'https',
 36      * ]
 37      * For example:
 38      * ```php
 39      * $baseDomain = 'example.com';
 40      * $subDomains = [
 41      *    '' => [
 42      *         'component' => [
 43      *              // `class` could be ignored as it is `vistart\Models\components\MultiDomainsUrlManager`.

 45      *             'enablePrettyUrl' => true,
 46      *             'showScriptName' => false,
 47      *             'suffix' => '.html',
 48      *             // the other properties...
 49      *         ],
 50      *         'scheme' => 'http', // `scheme` could be ignored as it is 'http'.
 51      *     ],
 52      *     'my' => [
 53      *         'component' => [
 54      *             'enablePrettyUrl' => true,
 55      *             'showScriptName' => false,
 56      *             'suffix' => '.html',
 57      *             'rules' => [
 58      *                 'posts/<year:\d{4}>/<category>' => 'post/index',
 59      *                 'posts' => 'post/index',
 60      *                 'post/<id:\d+>' => 'post/view',
 61      *             ],
 62      *         ],
 63      *         'scheme' => 'https',
 64      *     ],
 65      *     'sso' => [
 66      *         'component' => [
 67      *             'enablePrettyUrl' => true,
 68      *             'showScriptName' => false,
 69      *             'rules' => [
 70      *                 '' => 'sso/login',
 71      *                 'logout' => 'sso/logout',
 72      *             ],
 73      *         ],
 74      *         'scheme' => 'https',
 75      *     ],
 76      * ];
 77      * ```
 78      * and you can configure the `config/web.php` (basic template) or `config/main.php` (advanced template)
 79      * `components` section to add an element named `multiDomainsManager`:
 80      * ```php
 81      * $config = [
 82      *     ...
 83      *     'components' => [
 84      *         ...
 85      *         'multiDomainsManager' => [
 86      *             'baseDomain' => 'example.com',
 87      *             'subDomains' => [
 88      *                 'my' => [

 92      *             ],
 93      *         ],
 94      *         ...
 95      *     ],
 96      * ];
 97      * ```

100     public $subDomains = [];

108     /**
109      * Get UrlManager component of specified sub-domain application.
110      * @param string $subdomain
111      * @return \yii\web\UrlManager
112      */
113     public function get($subdomain = '')
114     {
115         if (!isset($this->subDomains[$subdomain])) {
116             return null;
117         }
118         $subDomainConfig = $this->subDomains[$subdomain];
119         if (!isset($subDomainConfig['component'])) {
120             return null;
121         }
122         if (!isset($subDomainConfig['component']['class'])) {
123             $subDomainConfig['component']['class'] = MultiDomainsUrlManager::className();
124         }
125         if (!isset($subDomainConfig['component']['hostInfo'])) {
126             if (!isset($subDomainConfig['scheme'])) {
127                 $subDomainConfig['scheme'] = 'http';
128             }
129             $subDomainConfig['component']['hostInfo'] = $subDomainConfig['scheme'] . // 'http' or 'https'
130                 "://" . // delimiter
131                 ($subdomain === '' ? '' : "$subdomain.") . // attach subdomain
132                 $this->baseDomain;                                               // base domain
133         }
134         return Yii::createObject($subDomainConfig['component']);
135     }
136 
137     /**
138      * Get URL Manager of current domain web application.
139      * @return MultiDomainsUrlManager url manager instance.
140      */
141     public function getCurrent()
142     {
143         return $this->get($this->currentDomain) ? : Yii::$app->urlManager;
144     }
