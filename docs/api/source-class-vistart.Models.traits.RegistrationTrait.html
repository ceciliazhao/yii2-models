 36     /**
 37      * Register new user.
 38      * It is equivalent to store the current user and its associated models into
 39      * database synchronously. The registration will be terminated immediately
 40      * if any errors occur in the process, and all the earlier steps succeeded
 41      * are rolled back.
 42      * If current user is not a new one(isNewRecord = false), the registration
 43      * will be skipped and return false.
 44      * The $eventBeforeRegister will be triggered before registration starts.
 45      * If registration finished, the $eventAfterRegister will be triggered. or
 46      * $eventRegisterFailed will be triggered when any errors occured.
 47      * @param array $associatedModels The models associated with user to be stored synchronously.
 48      * @return boolean Whether the registration succeeds or not.
 49      * @throws \yii\db\IntegrityException
 50      */
 51     public function register($associatedModels = []) {
 52         if (!$this->isNewRecord) {
 53             return false;
 54         }
 55         $this->trigger(static::$eventBeforeRegister);
 56         $transaction = $this->getDb()->beginTransaction();
 57         try {
 58             if (!$this->save()) {
 59                 throw new \yii\db\IntegrityException('Registration Error(s) Occured.', $this->errors);
 60             }
 61             foreach ($associatedModels as $model) {
 62                 if (!$model->save()) {
 63                     throw new \yii\db\IntegrityException('Registration Error(s) Occured.', $model->errors);
 64                 }
 65             }
 66             $transaction->commit();
 67         } catch (\yii\db\Exception $ex) {
 68             $transaction->rollBack();
 69             Yii::warning($ex->errorInfo, 'user\register');
 70             $this->trigger(static::$eventRegisterFailed);
 71             return $ex;
 72         }
 73         $this->trigger(static::$eventAfterRegister);
 74         return true;
 75     }
 76 
 77     /**
 78      * Deregister current uset itself.
 79      * It is equivalent to delete current user and its associated models. BUT it
 80      * deletes current user ONLY, the associated models will not be deleted
 81      * forwardly. So you should set the foreign key of associated models' table
 82      * referenced from primary key of user table, and their association mode is
 83      * 'on update cascade' and 'on delete cascade'.
 84      * @return boolean Whether deregistration succeeds or not.
 85      */
 86     public function deregister() {
 87         $this->trigger(static::$eventBeforeDeregister);
 88         $result = $this->delete();
 89         if ($result == 1) {
 90             $this->trigger(static::$eventAfterDeregister);
 91         } else {
 92             $this->trigger(static::$eventDeregisterFailed);
 93         }
 94         return $result == 1;
 95     }
