 17 /**
 18  * This trait allow its owner to enable the entity to be blamed by user.
 19  * @property-read boolean $isConfirmed
 20  * @property integer $confirmation
 21  * @property-read array $confirmationRules
 22  * @property string $confirmCode the confirm code used for confirming the content. 
 23  * You can disable this attribute and create a new model for storing confirm code as
 24  * its low-frequency usage.
 25  * @version 2.0
 26  * @author vistart <i@vistart.name>
 27  */
 28 trait ConfirmationTrait {

 45     /**
 46      * @var string This attribute specify the name of confirm_code attribute, if

 53      * @var integer The expiration in seconds. If $confirmCodeAttribute is

 68     public $initConfirmTime = '1970-01-01 00:00:00';

 73     /**
 74      * 
 75      * @return boolean
 76      * @throws \yii\base\NotSupportedException
 77      */
 78     public function applyConfirmation() {
 79         if (!$this->confirmCodeAttribute) {
 80             throw new \yii\base\NotSupportedException('This method is not implemented.');
 81         }
 82         $this->confirmCode = $this->generateConfirmationCode();
 83         if (!$this->save()) {
 84             return false;
 85         }
 86     }

 88     /**
 89      * 
 90      * @param string $code
 91      */
 92     public function setConfirmCode($code) {
 93         if (!$this->confirmCodeAttribute) {
 94             return;
 95         }
 96         $confirmCodeAttribute = $this->confirmCodeAttribute;
 97         $this->$confirmCodeAttribute = $code;
 98         if (!$this->confirmTimeAttribute) {
 99             return;
100         }
101         $confirmTimeAttribute = $this->confirmTimeAttribute;
102         if (!empty($code)) {
103             $this->$confirmTimeAttribute = date('Y-m-d H:i:s');
104             return;
105         }
106         $this->$confirmTimeAttribute = $this->initConfirmTime;
107     }
108 
109     /**
110      * 
111      * @return string
112      */
113     public function getConfirmCode() {
114         $confirmCodeAttribute = $this->confirmCodeAttribute;
115         return is_string($confirmCodeAttribute) ? $this->$confirmCodeAttribute : null;
116     }
117 
118     /**
119      * Confirm the current content.
120      * @param string $code
121      * @return boolean
122      */
123     public function confirm($code) {
124         if (!$this->confirmationAttribute || !$this->validateConfirmationCode($code)) {
125             return false;
126         }
127         $this->confirmation = self::$confirmTrue;
128         return $this->save();
129     }

131     /**
132      * 
133      * @return string
134      */
135     public function generateConfirmationCode() {
136         return substr(sha1(Yii::$app->security->generateRandomString()), 0, 8);
137     }

139     /**
140      * Validate the confirmation code.
141      * @param string $code
142      * @return boolean Whether the confirmation code is valid.
143      */
144     public function validateConfirmationCode($code) {
145         $confirmCodeAttribute = $this->confirmCodeAttribute;
146         if (!$confirmCodeAttribute)
147             return true;
148         return $this->$confirmCodeAttribute === $code;
149     }

151     /**
152      * Get confirmation status of current model.
153      * @return boolean Whether current model has been confirmed.
154      */
155     public function getIsConfirmed() {
156         $confirmationAttribute = $this->confirmationAttribute;
157         return is_string($confirmationAttribute) ? $this->$confirmationAttribute > static::$confirmFalse : true;
158     }

160     /**
161      * Initialize the confirmation status.
162      * This method is ONLY used for being triggered by event. DO NOT call,

164      * @param \yii\base\Event $event
165      */
166     public function onInitConfirmation($event) {
167         $sender = $event->sender;
168         if (!$sender->confirmationAttribute) {
169             return;

171         $sender->confirmation = self::$confirmFalse;
172         $sender->confirmCode = '';
173     }

175     /**
176      * 
177      * @param mixed $value
178      */
179     public function setConfirmation($value) {
180         $confirmationAttribute = $this->confirmationAttribute;
181         if (!$confirmationAttribute)
182             return;
183         $this->$confirmationAttribute = $value;
184         $this->trigger(self::$eventConfirmationChanged);
185     }

187     /**
188      * 
189      * @return mixed
190      */
191     public function getConfirmation() {
192         $confirmationAttribute = $this->confirmationAttribute;
193         return is_string($confirmationAttribute) ? $this->$confirmationAttribute : null;
194     }
195 
196     /**
197      * When confirmation status changed, this event will be triggered. If
198      * confirmation succeeded, the confirm_time will be assigned to current time,
199      * or the confirm_time will be assigned to initConfirmTime.
200      * This method is ONLY used for being triggered by event. DO NOT call,
201      * override or modify it directly, unless you know the consequences.
202      * @param \yii\base\Event $event
203      */
204     public function onConfirmationChanged($event) {
205         $sender = $event->sender;
206         $confirmationAttribute = $sender->confirmationAttribute;
207         if (!$confirmationAttribute)
208             return;
209         if ($sender->isAttributeChanged($confirmationAttribute)) {
210             $sender->confirmCode = '';
211             if ($sender->$confirmationAttribute == self::$confirmFalse) {
212                 $sender->trigger(self::$eventConfirmationCanceled);
213                 return;
214             }
215             $sender->trigger(self::$eventConfirmationSuceeded);
216             $sender->resetOthersConfirmation();
217         }
218     }
219 
220     /**
221      * 
222      * @return array
223      */
224     public function getConfirmationRules() {
225         if (!$this->confirmationAttribute) {
226             return [];
227         }
228         return [
229             [[$this->confirmationAttribute], 'number', 'integerOnly' => true, 'min' => 0],
230             [[$this->confirmTimeAttribute], 'safe'],
231         ];
232     }
233 
234     /**
235      * When the content changed, reset confirmation status.
236      */
237     protected function resetConfirmation() {
238         $contentAttribute = $this->contentAttribute;
239         if (!$contentAttribute)
240             return;
241         if (is_array($contentAttribute)) {
242             foreach ($contentAttribute as $attribute) {
243                 if ($this->isAttributeChanged($attribute)) {
244                     $this->confirmation = self::$confirmFalse;
245                     break;
246                 }
247             }
248         } elseif ($this->isAttributeChanged($contentAttribute)) {
249             $this->confirmation = self::$confirmFalse;
250         }
251     }
252 
253     /**
254      * Reset others' confirmation when the others own the same content.
255      */
256     protected function resetOthersConfirmation() {
257         if (!$this->confirmationAttribute || empty($this->userClass))
258             return;
259         $contents = self::find()->where([$this->contentAttribute => $this->content])->andWhere(['not', $this->createdByAttribute, $this->creator])->all();
260         foreach ($contents as $content) {
261             $content->confirmation = self::$confirmFalse;
262             $content->save();
263         }
264     }
